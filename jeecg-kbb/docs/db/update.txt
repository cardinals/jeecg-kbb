/*配件资料js增强*/
$(function(){
if(location.href.indexOf("goAdd.do")!=-1){
	$("#fnumber").css("readonly","readonly");
	var baseUrl=$('base').attr("href");
	$.ajax({  
		url: baseUrl+"baseServiceController.do?getStandardBillNo",  
		type: "get",  
		dataType: "json",  
		success: function (result) {
			if(result.success){
				$("#fnumber").val(result.msg);
			}				
		}
	});
}
});
